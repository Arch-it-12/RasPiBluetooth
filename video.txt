# Get the latest Raspbian packages
sudo apt-get update
sudo apt-get upgrade

# Get the latest firmware
sudo rpi-update

# Get the source code for the V4L utilities
git clone git://git.linuxtv.org/v4l-utils.git
cd v4l-utils

# Install some packages needed to build it
sudo apt-get install autoconf gettext libtool libjpeg62-dev vlc

# Configure and build it.
./bootstrap.sh # RASPI SE EDIT
./configure
make
sudo make install
